from ultralytics import YOLO
import cv2
import time
from pymodbus.client import ModbusSerialClient as ModbusClient

# --- RS485 / MODBUS ASCII 7-BIT GÜNCEL AYARLARI ---
# HMI ekranın 'Communication Error' hatasına düşmesini engellemek için 
# timeout süresi 0.05 saniyeye (ışık hızına) düşürüldü.
client = ModbusClient(
    port='COM8',                 # Çeviricinin bilgisayardaki güncel COM portu
    baudrate=9600,               # PLC ile aynı olmalı
    bytesize=7,                  # ASCII haberleşme için 7 bit
    parity='E',                  # Even (Çift) parite
    stopbits=1,                  # 1 stop biti
    timeout=0.05,                # Python hattan çok hızlı çıkıp kapıyı HMI'a bırakacak
    retries=3,                   # Paket kaçarsa 3 kez dener
    framer='ascii'               # Delta PLC varsayılan ASCII modu
)

# Kütüphane versiyon farklarını çözen istasyon numarası tanımı
client.default_slave = 1

def plc_hata_sinyali_gonder(state):
    try:
        if not client.connected:
            client.connect()
        
        # M105 adresinin Modbus ASCII karşılığı: 2153
        result = client.write_coil(2153, state)
        
        # HMI çakışması anında ana döngünün (kameraların) donmasını engeller
        if result.isError():
            pass
    except Exception:
        pass
    finally:
        # Sinyal gider gitmez hattı HMI ekranın kullanması için DERHAL serbest bırakıyoruz
        client.close()

# --- YOLO MODEL YÜKLEME ---
model = YOLO(r"C:\Users\ulker\Desktop\yolo\yolo\best.pt")

# Kamera Ayarları
cap = cv2.VideoCapture(0)
cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

last_signal_time = 0
son_gonderilen_durum = None  # Sürekli aynı sinyali gönderip hattı şişirmeyi engeller

print("Sistem Başlatıldı. RS485 Kararlı ASCII (HMI Paylaşımlı) Modda PLC izleniyor...")

while True:
    ret, frame = cap.read()
    if not ret:
        break

    # Görüntü İşleme (YOLO Yapay Zeka Modeli)
    results = model(frame, device=0, imgsz=640, conf=0.25, half=True, verbose=False)
    
    hata_var_mi = False
    for r in results:
        # Eğer ekranda herhangi bir hata/defo kutusu çizildiyse
        if len(r.boxes) > 0:
            hata_var_mi = True
            break

    current_time = time.time()
    
    # --- Akıllı Sinyal Gönderme Mantığı ---
    # Sadece durum değiştiğinde (0'dan 1'e veya 1'den 0'a geçişte) SADECE 1 KEZ yazar.
    # Bu sayede RS485 hattındaki gereksiz trafik %90 oranında azalır.
    if hata_var_mi and son_gonderilen_durum != True:
        if current_time - last_signal_time > 0.5:
            print("Hata (Defo) Tespit Edildi! PLC'ye (M105) SET sinyali gönderiliyor...")
            plc_hata_sinyali_gonder(True)
            son_gonderilen_durum = True
            last_signal_time = current_time
            
    elif not hata_var_mi and son_gonderilen_durum != False:
        if current_time - last_signal_time > 0.5:
            # Hata durumu bittiyse PLC'deki hata bitini temizle/sıfırla
            plc_hata_sinyali_gonder(False)
            son_gonderilen_durum = False
            last_signal_time = current_time

    # Ekranda Çizim ve Görselleştirme
    annotated = results[0].plot()
    cv2.imshow("YOLO Kalite Kontrol - KOCAKER", annotated)

    # 'q' tuşuna basılırsa döngüden çık
    if cv2.waitKey(1) & 0xFF == ord("q"):
        break

cap.release()
cv2.destroyAllWindows()
